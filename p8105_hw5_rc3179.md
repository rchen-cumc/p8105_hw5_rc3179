fill_missing = function(x) {
  if (is.numeric(x)) {
    x = x %>% replace_na(mean(x, na.rm = TRUE))
  } else if (is.character(x)) {
    x = x %>% replace_na("virginica")
  } else {
    stop("x is neither numeric nor a character")
  }
}

iris_filled =
  map(iris_with_missing, fill_missing) %>% 
  bind_rows()
